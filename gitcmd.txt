创建一个新仓库

touch README.md

推送现有文件夹
cd existing_folder
git init
git remote add origin git@gitcode.net:weixin_37548620/bitanalyzer.git
git add .
git commit -m "Initial commit"
git push -u origin master

推送现有的 Git 仓库

git remote add origin git@gitcode.net:weixin_37548620/bitanalyzer.git
git push -u origin --all
git push -u origin --tags
